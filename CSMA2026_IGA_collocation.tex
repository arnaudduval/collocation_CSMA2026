\begin{frame}
  \frametitle{Modélisation B-Spline}
  \begin{columns}
    \begin{column}{0.65 \linewidth}
      \begin{itemize}
        \item Représentation des objets géométriques sous forme {\bf paramétrique}
        \item Exemple d'une courbe dans un plan
        \begin{itemize}
          \item Expression générale
          \begin{equation*}
            C(\xi) = \left( x(\xi), y(\xi) \right) \quad \forall \xi \in \mathcal{U}
          \end{equation*}
          \item B-splines
          \begin{equation*}
            C(\xi) = \sum_{i=1}^{n} N_i(\xi) P_i \quad \forall \xi \in [0, 1]
          \end{equation*}
          \item IGA
          \begin{equation*}
            {\bf U}(\xi) = \sum_{i=1}^{n} N_i(\xi) U_i \quad \forall \xi \in [0, 1]
          \end{equation*}
        \end{itemize}
      \end{itemize}
    \end{column}
    \begin{column}{0.35\linewidth}
      \begin{center}

      \end{center}
    \end{column}
  \end{columns}
\end{frame}

        \item<1-> Condition de couplage mortar
        \begin{equation}
          \int_{\Omega_1 \bigcap \Omega_2} \vec{\lambda} \left( \vec{u_1} - \vec{u_2} \right) \text{d}V = 0
        \end{equation}

          \sum_{k=1}^n \vec{\lambda} \left( x_k \right) \left( \vec{u_1} \left(x_k\right) - \vec{u_2} \left( x_k \right) \right) = 0
        \end{equation}

\begin{frame}
  \frametitle{Méthode hybride mortar/collocation}
  \begin{columns}
    \begin{column}{0.6\linewidth}
      \begin{itemize}
        \item IGA : approximation de la solution avec une continuité élevée
        \item Développement de Taylor à l'ordre 1 pour $u_i$ et $\lambda$
        \begin{equation}
          u_i \left(x \right) \approx u_i \left( x_0 \right) + \frac{\partial u_i}{\partial x} \left( x_0 \right) \left( x - x_0 \right)
        \end{equation}
        \begin{equation}
          \lambda \left(x \right) \approx \lambda \left( x_0 \right) + \frac{\partial \lambda}{\partial x} \left( x_0 \right) \left( x - x_0 \right)
        \end{equation}
        \item Intégration sur des disques $\omega_k$ de diamètre $d$
        \only<1>{
        \begin{equation}
          \sum_{k=1}^n \int_{\omega_k} \vec{\lambda} \left( x_k \right) \left( \vec{u_1} \left( x_k \right) - \vec{u_2} \left(x_k \right) \right) \text{d} \omega_k = 0
        \end{equation}
        }
        \only<2>{
        \begin{equation}
          \sum_{k=1}^n \vec{\lambda} \left( x_k \right) \left( \vec{u_1} \left( x_k \right) - \vec{u_2} \left( x_k \right)\right) + \frac{d^2}{8} \nabla \vec{\lambda} \left( x_k \right)  : \left( \nabla \vec{ u_1} \left( x_k \right) - \nabla \vec{u_2} \left( x_k \right)\right) = 0
        \end{equation}
        }
        \only<3>{
        \begin{equation}
          \sum_{k=1}^n {\color{red} \vec{\lambda} \left( x_k \right) \left( \vec{u_1} \left( x_k \right) - \vec{u_2} \left( x_k \right)\right) } + \kappa {\color{blue} \nabla \vec{\lambda} \left( x_k \right) : \left( \nabla \vec{u_1} \left( x_k \right) - \nabla \vec{u_2} \left( x_k \right)\right) } = 0
        \end{equation}
        }
      \end{itemize}
    \end{column}
    \begin{column}{0.39\linewidth}

\end{frame}

\begin{frame}
  \frametitle{Méthode hybride mortar/collocation}
  \begin{itemize}
    \item Pas besoin de paramétrisation de la zone recouvrante
    \item Paramétrisation du patch support des multiplicateur de Lagrange (qui peut être plus grand que la zone recouvrante \faArrowRight{} simple)
    \item Pondération de la raideur dans la zone recouvrante
    \begin{itemize}
      \item $\times \frac{1}{2}$ aux points d'intégration situés dans $\Omega_1 \bigcap \Omega_2$
      \item Avec une fonction plus \textit{smooth} (mais on veux éviter une paramétrisation de l'intersection)
    \end{itemize}
    \item \faWarning{} Beaucoup d'inversions de paramétrisation
    \item Résolution avec un seul système linéaire monolithique ou avec un solveur itératif FETI/IETI
    \begin{equation}
      \begin{bmatrix}
        K^{(1)} & 0 & C^{(1)T} \\
        0 & K^{(2)} & C^{(2)T} \\
        C^{(1)} & C^{(2)} & 0
      \end{bmatrix}
      \begin{bmatrix}
        u^{(1)} \\ u^{(2)} \\ \lambda
      \end{bmatrix}
      =
      \begin{bmatrix}
        F^{(1)} \\ F^{(2)} \\ 0
      \end{bmatrix}
    \end{equation}
  \end{itemize}
\end{frame}
